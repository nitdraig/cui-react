import { useEffect, useState } from "react";
import { Button } from "cui-react";
import { useRouter } from 'next/router';

export default function Home() {
  const router = useRouter();
  const [stars, setStars] = useState(null);

  useEffect(() => {
    const fetchStars = async () => {
      try {
        const response = await fetch('https://api.github.com/repos/nitdraig/cui-react');
        const data = await response.json();
        setStars(data.stargazers_count);
      } catch (error) {
        console.error('Error fetching GitHub stars:', error);
      }
    };

    fetchStars();
  }, []);

  return (
    <section className="flex flex-col md:flex-row items-center justify-center lg:h-screen h-full bg-gray-500 dark:bg-[#171717]">
      <div className="md:w-1/2 text-center md:text-left px-8 md:px-16">
        <img src="logo.png" className="lg:w-32 w-28 lg:mt-0 mt-4 shadow-lg p-2" alt="CUI-Logo"/>
        <h1 className="text-4xl font-bold text-gray-900 dark:text-white mb-4">Start developing easily, quickly and for free.</h1>
        <p className="text-lg text-gray-700 dark:text-gray-300 mb-6">Collection of React components designed with Tailwind CSS. Find detailed guides, practical examples and best practices to integrate these components into your projects quickly and efficiently.</p>
        <div className="flex items-center mb-4">
          {stars !== null ? (
            <div className="flex items-center text-lg text-gray-700 dark:text-gray-300">
           <span>‚≠ê {stars} {stars === 1 ? "star" : "stars"} on GitHub</span>
              <img src="github.svg" className="w-6 h-6 ml-2" alt="GitHub logo"/>
            </div>
          ) : (
            <p className="text-lg text-gray-700 dark:text-gray-300">Loading stars...</p>
          )}
        </div>
        <Button text="Get started" size="large" color="primary" onClick={() => router.push('/docs')} />
      </div>
      <div className="md:w-1/2 md:px-8">
        <div className="aspect-w-16 aspect-h-9">
          <video className="mt-6 rounded-xl border dark:border-zinc-800" muted autoPlay playsInline loop>
            <source src="/intro.mp4" type="video/mp4" />
            Your browser does not support the video tag.
          </video>
        </div>
      </div>
    </section>
  );
}
